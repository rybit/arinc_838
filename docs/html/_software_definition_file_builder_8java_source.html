00074         public void initialize(BdfFile file) throws IOException {
00075 
00076                 file.seek(BdfFile.FILE_FORMAT_VERSION_LOCATION);
00077                 byte[] fileFormatVersion = file.readHexbin32();
00078                 if (!Arrays.equals(fileFormatVersion,
00079                                 SoftwareDefinitionFileBuilder.DEFAULT_FILE_FORMAT_VERSION)) {

00081                                         "File format not recognized. Expected: "
00082                                                         + SoftwareDefinitionFileBuilder.DEFAULT_FILE_FORMAT_VERSION
00083                                                         + " Got: " + fileFormatVersion);
00084                 }
00085 
00086                 file.seek(file.readSoftwareDescriptionPointer());
00087                 softwareDescription = new SoftwareDescriptionBuilder(file);
00088 
00089                 file.seek(file.readTargetDefinitionsPointer());
00090                 long targetHardwareCount = file.readUint32();
00091                 for (int i = 0; i < targetHardwareCount; i++) {
00092                         long nextHardware = file.readUint32();
00093                         TargetHardwareDefinitionBuilder hardware = new TargetHardwareDefinitionBuilder(
00094                                         file);
00095                         this.getTargetHardwareDefinitions().add(hardware);
00096                         file.seek(nextHardware);
00097                 }
00098 
00099                 file.seek(file.readFileDefinitionsPointer());
00100                 long fileDefinitionCount = file.readUint32();
00101                 for (int i = 0; i < fileDefinitionCount; i++) {
00102                         long nextFile = file.readUint32();
00103                         FileDefinitionBuilder fileDefinition = new FileDefinitionBuilder(
00104                                         file);
00105                         this.getFileDefinitions().add(fileDefinition);
00106                         file.seek(nextFile);
00107                 }
00108 
00109                 file.seek(file.readSdfIntegrityDefinitionPointer());
00110                 this.setSdfIntegrityDefinition(new IntegrityDefinitionBuilder(file));
00111

00177         @Override
00178         public int buildBinary(BdfFile file) throws IOException {
00179                 file.seek(0);
00180                 // write the header
00181                 file.writePlaceholder(); // file size
00182                 file.writeHexbin32(getFileFormatVersion());
00183                 file.writePlaceholder(); // software description pointer
00184                 file.writePlaceholder(); // target hardware definition pointer
00185                 file.writePlaceholder(); // file definition pointer
00186                 file.writePlaceholder(); // SDF integrity definition pointer
00187                 file.writePlaceholder(); // LSP integrity definition pointer
00188                 this.getSoftwareDescription().buildBinary(file);
00189 
00190                 // Write the target hardware definitions
00191                 int size = this.getTargetHardwareDefinitions().size();
00192                 file.writeTargetDefinitionsPointer();
00193                 file.writeUint32(size);
00194                 if (size > 0) {
00195                         this.getTargetHardwareDefinitions().get(size - 1).setIsLast(true);
00196                         for (int i = 0; i < size; i++) {
00197                                 this.getTargetHardwareDefinitions().get(i).buildBinary(file);
00198                         }
00199                 }
00200 
00201                 // write the file definitions
00202                 size = this.getFileDefinitions().size();
00203                 file.writeFileDefinitionsPointer();
00204                 file.writeUint32(size);
00205                 this.getFileDefinitions().get(size - 1).setIsLast(true);
00206                 for (int i = 0; i < size; i++) {
00207                         this.getFileDefinitions().get(i).buildBinary(file);
00208                 }
00209 
00210                 // write the SDF integrity def
00211                 file.writeSdfIntegrityDefinitionPointer();
00212                 this.getSdfIntegrityDefinition().setIntegrityValue(
00213                                 Converter.hexToBytes("0000000A"));
00214                 this.getSdfIntegrityDefinition().buildBinary(file);
00215 
00216                 // write the LSP integrity def
00217                 file.writeLspIntegrityDefinitionPointer();
00218                 this.getLspIntegrityDefinition().setIntegrityValue(
00219                                 Converter.hexToBytes("0000000A"));
00220                 this.getLspIntegrityDefinition().buildBinary(file);
00221 
00222                 // write the file size
00223                 file.seek(0);
00224                 file.writeUint32(file.length());
00225                 file.seek(file.length());
00226 
00227                 return (int) file.length();
